title: "Incorporation Completed"
---

### Understanding the Event

The `incorporation.completed` event is triggered automatically when an incorporation data has been verified and confirmed by our internal team.

### Payload Fields

<ResponseField name="webhook_id" type="string">
  A unique identifier for this webhook in a **UUID** format.\
  **Example**: `123e4567-e89b-12d3-a456-426614174000`
</ResponseField>

<ResponseField name="event" type="string">
  The name of the event that triggered this webhook.\
  **Value**: `incorporation.completed`
</ResponseField>

<ResponseField name="message" type="string">
  Provides a human-readable message indicating the result of the webhook event.\
  **Value**: `The incorporation has been completed.`
</ResponseField>

<ResponseField name="incorporation_id" type="string">
  The unique identifier for the incorporation in a **UUID** format.\
  **Example**: `123e4567-e89b-12d3-a456-426614174000`
</ResponseField>

<ResponseField name="type" type="string">
  The legal name of the company.\
  **Example**: `Beglaubigt.de GmbH`
</ResponseField>

<ResponseField name="purpose" type="string">
  The purpose or business objective of the company.\
  **Example**: `Digital Notarization Services for Germany and Europe`
</ResponseField>

<ResponseField name="capital" type="number">
  The total share capital of the company in EUR.\
  **Example**: `25000`
</ResponseField>

<ResponseField name="address" type="object">
  The address of the company.

  <Expandable title="Address">
    <ResponseField name="street" type="string">
      The street name and number of the company.\
      **Example**: `Schillerstraße 5`
    </ResponseField>
    <ResponseField name="street_second_line" type="string">
      Additional address information, such as apartment number, suite, floor, or building.\
      **Example**: `Apartment 4B`
    </ResponseField>
    <ResponseField name="zip" type="string">
      The postal code of the company.\
      **Example**: `80336`
    </ResponseField>
    <ResponseField name="city" type="string">
      The city of the company.\
      **Example**: `Munich`
    </ResponseField>
    <ResponseField name="country" type="string">
      The country of the company.\
      **Example**: `Germany`
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="status" type="string">
  The status of the incorporation.\
  **Example**: `Initiated`, `Pending`, `Appointment Confirmed` or `Completed`
</ResponseField>

<ResponseField name="url" type="string">
  The URL of the incorporation.\
  **Example**: `https://app.beglaubigt.de/incorporate/acme-holding-gmbh/123e4567-e89b-12d3-a456-426614174000`
</ResponseField>

<ResponseField name="documents" type="object">
  A collection of documents related to the company.

  <Expandable title="Documents">
    <ResponseField name="articles_of_association" type="array">
      The articles of association for the company.

      <Expandable title="Article of Association">
        <ResponseField name="id" type="string">
          The unique identifier for the document in a UUID format.\
          **Example**: `07347a09-7fa2-434f-a5f1-ec897f16d4b9`
        </ResponseField>
        <ResponseField name="type" type="string">
          The type of document as specified in the request.\
          **Example**: `Articles of Association`
        </ResponseField>
        <ResponseField name="status" type="string">
          Indicates the current processing state of the document.\
          **Possible values**: `Initiated` or `Completed`
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="directors" type="array">
  An array of directors managing the company.
  <Expandable title="Director">
    <ResponseField name="first_name" type="string">
      First name of the director.\
      **Example**: `Maria`
    </ResponseField>
    <ResponseField name="last_name" type="string">
      Last name of the director.\
      **Example**: `Weber`
    </ResponseField>
    <ResponseField name="email" type="string">
      Email address of the director.\
      **Example**: `maria.weber@example.com`
    </ResponseField>
    <ResponseField name="phone" type="string">
      The phone number of the director.\
      **Example**: `+491234567890`
    </ResponseField>
    <ResponseField name="dob" type="string">
      Date of birth of the director in `YYYY-MM-DD` format.\
      **Example**: `1978-03-15`
    </ResponseField>
    <ResponseField name="address" type="object">
      Address of the director.
      <Expandable title="Address">
        <ResponseField name="street" type="string">
          The street name and number of the director.\
          **Example**: `Schillerstraße 5`
        </ResponseField>
        <ResponseField name="street_second_line" type="string">
          Additional address information, such as apartment number, suite, floor, or building.\
          **Example**: `Apartment 4B`
        </ResponseField>
        <ResponseField name="zip" type="string">
          The postal code of the director.\
          **Example**: `80336`
        </ResponseField>
        <ResponseField name="city" type="string">
          The city of the director.\
          **Example**: `Munich`
        </ResponseField>
        <ResponseField name="country" type="string">
          The country of the director.\
          **Example**: `Germany`
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="authority" type="string">
      The director’s signing authority.\
      **Possible values**: `sole` or `joint`
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="documents" type="object">
  A collection of documents related to the company.
  <Expandable title="Documents">
    <ResponseField name="articles_of_association" type="array">
      The articles of association for the company.
      <Expandable title="Article of Association">
        <ResponseField name="id" type="string">
          The unique identifier for the document in a UUID format.\
          **Example**: `07347a09-7fa2-434f-a5f1-ec897f16d4b9`
        </ResponseField>
        <ResponseField name="type" type="string">
          The type of document as specified in the request.\
          **Example**: `Articles of Association`
        </ResponseField>
        <ResponseField name="status" type="string">
          Indicates the current processing state of the document.\
          **Possible values**: `Initiated` or `Completed`

      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="notarization" type="object">
  Details about the notarization process.
  <Expandable title="Notarization">
    <ResponseField name="type" type="string">
      The method of notarization.\
      **Possible values**: `offline` or `online`
    </ResponseField>
    <ResponseField name="express" type="boolean">
      Requests expedited notarization.
    </ResponseField>
    <ResponseField name="date" type="string">
      The notarization date in `YYYY-MM-DD` format.\
      **Example**: `2025-04-15`
    </ResponseField>
    <ResponseField name="location" type="string">
      The location of the notarization.\
      **Example**: `Munich`
    </ResponseField>
    <ResponseField name="notary" type="object">
      Details about the notary.
      <Expandable title="Notary">
        <ResponseField name="first_name" type="string">
          The first name of the notary.\
          **Example**: `John`
        </ResponseField>
        <ResponseField name="last_name" type="string">
          The last name of the notary.\
          **Example**: `Doe`
        </ResponseField>
        <ResponseField name="email" type="string">
          The email address of the notary.\
          **Example**: `john.doe@example.com`
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="client_reference" type="string">
  Your custom reference string provided during incorporation creation.\
  **Example**: `REF-2025-001`
</ResponseField>

<ResponseField name="timestamp" type="string">
  The exact date and time when the webhook was triggered, formatted as an **ISO 8601** string in UTC.\
  **Example**: `2025-06-01T08:00:00.443Z`
</ResponseField>
