---
title: "Create Notarization"
api: "POST https://app.platus.com/api/notarization"
description: "This endpoint creates a notarization request for a document"
---

### Authorization

<ParamField header="Authorization" type="string" required>
  Bearer authentication header of the form `Bearer <token>`, where `<token>` is your auth token.
</ParamField>

### Body

<ParamField body="jurisdiction" type="string" required>
  The jurisdiction in which the notarization will be executed. Example: "US"
</ParamField>

<ParamField body="state" type="string">
  The state or region within the jurisdiction. Example: "CA"
</ParamField>

<ParamField body="documents" type="array" required>
  An array of documents to be notarized. Each document contains document_type, document_file, is_already_signed, and signers information.
</ParamField>

<ParamField body="client_reference" type="string">
  A custom client reference string for tracking purposes. Example: "REF-2024-001"
</ParamField>

<ParamField body="client_postback_url" type="string">
  A URL where status updates will be sent. Example: "https://client-domain.com/webhook-endpoint"
</ParamField>

### Response

<ResponseField name="case_id" type="string">
  The unique identifier for the created notarization case.
</ResponseField>

<ResponseField name="status" type="string">
  The status of the notarization request. Example: "created"
</ResponseField>

<ResponseField name="documents" type="array">
  An array of documents with metadata about their processing state.

<Expandable title="Toggle Document Details">

<ResponseField name="document_type" type="string">
  The type of document. Example: "Power of Attorney"
</ResponseField>

<ResponseField name="document_file" type="object">
  File details for the document, including type and url.
</ResponseField>

<ResponseField name="is_already_signed" type="boolean">
  Whether the document is already signed before notarization.
</ResponseField>

<ResponseField name="signers" type="array">
  Array of signer objects containing email, first_name, last_name, date_of_birth, and more.

<Expandable title="Toggle Signer Details">

<ResponseField name="email" type="string">
  Email address of the signer.
</ResponseField>

<ResponseField name="first_name" type="string">
  First name of the signer.
</ResponseField>

<ResponseField name="last_name" type="string">
  Last name of the signer.
</ResponseField>

<ResponseField name="date_of_birth" type="string">
  Date of birth of the signer in YYYY-MM-DD format.
</ResponseField>

<ResponseField name="labels" type="array">
  Label information for signature positioning or other metadata.
</ResponseField>

</Expandable>

</ResponseField>

</Expandable>

</ResponseField>
