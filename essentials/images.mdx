---
title: "Get Notarization"
api: "GET https://api.platus.com/v1/notarization/{case_id}"
description: "This endpoint retrieves the details of a notarization case"
---

### Authorization

<ParamField header="Authorization" type="string" required>
  Bearer authentication header of the form `Bearer <token>`, where `<token>` is your auth token.
</ParamField>

### Path Parameters

<ParamField path="case_id" type="string" required>
  The unique identifier of the notarization case to retrieve. Example: "case\_abc123xyz789"
</ParamField>

### Response

<ResponseField name="case_id" type="string">
  The unique identifier of the notarization case.
</ResponseField>

<ResponseField name="case" type="string">
  The type of case. Value: "notarization"
</ResponseField>

<ResponseField name="jurisdiction" type="string">
  The jurisdiction where the notarization was executed.
</ResponseField>

<ResponseField name="status" type="string">
  Current status of the notarization. Possible values: "created", "processing", "completed"
</ResponseField>

<ResponseField name="completed_at" type="string">
  ISO 8601 timestamp indicating when the notarization was completed.
</ResponseField>

<ResponseField name="client_reference" type="string">
  The client reference string provided during creation.
</ResponseField>

<ResponseField name="documents" type="array">
  List of documents included in this notarization.
</ResponseField>

<ResponseField name="documents.document_id" type="string">
  Unique identifier for the document.
</ResponseField>

<ResponseField name="documents.document_type" type="string">
  The type of document.
</ResponseField>

<ResponseField name="documents.document_url" type="string">
  URL to download the notarized document.
</ResponseField>

<ResponseField name="documents.document_hash" type="string">
  SHA-256 hash of the notarized document for verification.
</ResponseField>

<ResponseField name="documents.signers" type="array">
  List of signers for this document.
</ResponseField>

<ResponseField name="documents.signers.signer_id" type="string">
  Unique identifier for the signer.
</ResponseField>

<ResponseField name="documents.signers.email" type="string">
  Email address of the signer.
</ResponseField>

<ResponseField name="documents.signers.first_name" type="string">
  First name of the signer.
</ResponseField>

<ResponseField name="documents.signers.middle_name" type="string">
  Middle name of the signer (if provided).
</ResponseField>

<ResponseField name="documents.signers.last_name" type="string">
  Last name of the signer.
</ResponseField>

<ResponseField name="documents.signers.birthdate" type="string">
  Date of birth of the signer in YYYY-MM-DD format.
</ResponseField>

<ResponseField name="documents.signers.phone" type="string">
  Phone number of the signer.
</ResponseField>

<ResponseField name="documents.signers.verified" type="boolean">
  Indicates whether the signer's identity has been verified.
</ResponseField>
